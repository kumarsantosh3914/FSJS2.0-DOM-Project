  element.addEventListener("click", () => {
    let para = element.nextElementSibling;
    if (para.style.display === "block") {
      para.style.display = "none";
    } else {
      para.style.display = "block";
    }
    // para.style.color = "#758283";
  });
  element.style.backgroundColor = "#dadaf8";
});
